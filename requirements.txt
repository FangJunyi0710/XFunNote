# ===================== 核心引擎 =====================

typer               # 命令行工具（cli.py）
python-dotenv       # 环境变量管理（config.py）
future-uuid         # uuid7 生成
langchain           # AI Agent 框架
langchain-anthropic # Anthropic / DeepSeek 兼容封装
langchain-core      # LangChain 核心抽象
pydantic            # 数据模型与校验

fastapi             # RESTful API（backend/）
uvicorn[standard]   # ASGI 服务器

# ===================== 测试 =====================

pytest              # 测试框架
pytest-cov          # 测试覆盖率
